.MODEL SMALL
.STACK 100H
.DATA
    MSG DB 'ZERO$'
    MSG2 DB 'NOT ZERO$'
.CODE

MAIN PROC
    MOV AX,@DATA
    MOV DS, AX

    MOV AH,01H
    INT 21H
    MOV BL,AL
    SUB BL,'0'

    MOV AH,02H
    MOV DL, 13
    INT 21H
    MOV DL,10
    INT 21H
    
    CMP BL,0
    JZ ZR
    JMP NZR
ZR:
    LEA DX, MSG
    MOV AH,09H
    INT 21H
    JMP SESH
NZR:
    LEA DX, MSG2
    MOV AH, 09H
    INT 21H
        
SESH:
    MOV AH,4CH
    INT 21H

    MAIN ENDP
END MAIN
